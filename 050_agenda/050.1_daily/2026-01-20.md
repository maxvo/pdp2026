```dataviewjs
let keyDeliveryDuration = {};

// First pass: Collect all durations
dv.current().file.tasks.map(task => {
    const timeRange = task.text.match(/(\d{2}):(\d{2}) - (\d{2}):(\d{2})/);
    if (!timeRange) return null;
    
    const startMinutes = parseInt(timeRange[1]) * 60 + parseInt(timeRange[2]);
    const endMinutes = parseInt(timeRange[3]) * 60 + parseInt(timeRange[4]);
    const duration = endMinutes - startMinutes;

    const keyDeliveryMatch = task.text.match(/\[\[([^\]]+)\]\]/);
    const keyDelivery = keyDeliveryMatch ? keyDeliveryMatch[1] : "Entrega não cadastrada"

    if (!keyDeliveryDuration[keyDelivery]) {
        keyDeliveryDuration[keyDelivery] = 0;
    }
    keyDeliveryDuration[keyDelivery] += duration;
});

// Calculate total time across all key deliveries
const totalTime = Object.values(keyDeliveryDuration).reduce((sum, duration) => sum + duration, 0);

// Prepare table with percentage calculation
let table = [];
for (const [keyDelivery, totalDuration] of Object.entries(keyDeliveryDuration)) {
    const percentage = totalDuration / totalTime * 100;
    table.push([
        keyDelivery, 
        totalDuration,
        `${percentage.toFixed(1)}%`  // Format to 1 decimal place
    ]);
}

table.sort((a, b) => b[1] - a[1]);  // Sort by duration descending

// Create table with new percentage column
dv.table(
    ["Key Delivery", "Total Minutes", "% of Total"], 
    table
);
```
